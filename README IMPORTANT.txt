"message": "!setvip5 @customapi@[http://yourhost.com/vip.php?server=yourserverip&secret=yoursecretcode&user=@target@[1]&vip=@target@[2]&days=@target@[3]]"

server=yourserverip = the server with the running deepbot
secret=yoursecretcode = Your Secret API Code
